    <script>
        let currentMode = 'app';
        const terminal = document.getElementById('terminal');
        const resultArea = document.getElementById('result-area');

        function setMode(m) {
            currentMode = m;
            document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
            event.target.classList.add('active');
        }

        async function startRealGeneration() {
            const apiKey = document.getElementById('api-key-input').value;
            const prompt = document.getElementById('prompt-input').value;

            if (!apiKey) {
                showToast("Veuillez entrer votre clé API !");
                return;
            }
            if (!prompt) {
                showToast("Décrivez votre projet.");
                return;
            }

            // UI Updates
            terminal.style.display = 'block';
            terminal.innerText = "Connexion aux serveurs OpenAI...\n";
            resultArea.style.display = 'none';

            try {
                // Construction du prompt pour l'IA
                let systemPrompt = "";
                if(currentMode === 'app') systemPrompt = "Tu es un expert en développement mobile. Génère le code source (React Native ou Flutter) pour une application basée sur cette demande :";
                if(currentMode === 'site') systemPrompt = "Tu es un expert web. Génère le code HTML/CSS complet pour ce site :";
                if(currentMode === 'video') systemPrompt = "Tu es un expert en marketing viral. Donne un script détaillé pour une vidéo TikTok/Shorts sur ce sujet :";

                const response = await fetch('https://api.openai.com/v1/chat/completions', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': `Bearer ${apiKey}`
                    },
                    body: JSON.stringify({
                        model: "gpt-3.5-turbo", // ou gpt-4 si votre clé y a accès
                        messages: [
                            { role: "system", content: systemPrompt },
                            { role: "user", content: prompt }
                        ],
                        max_tokens: 1000
                    })
                });

                const data = await response.json();

                if (data.error) {
                    terminal.innerText = "ERREUR API : " + data.error.message;
                    terminal.style.color = "red";
                } else {
                    const aiCode = data.choices[0].message.content;
                    typeWriterEffect(aiCode);
                }

            } catch (error) {
                terminal.innerText = "Erreur de connexion : " + error.message;
                terminal.style.color = "red";
            }
        }

        // Effet machine à écrire pour le résultat
        function typeWriterEffect(text) {
            terminal.innerText = "";
            let i = 0;
            const speed = 5; // Vitesse d'affichage
            
            function type() {
                if (i < text.length) {
                    terminal.innerText += text.charAt(i);
                    terminal.scrollTop = terminal.scrollHeight;
                    i++;
                    setTimeout(type, speed);
                } else {
                    resultArea.style.display = 'block';
                    // Vérifier si c'est la première génération pour proposer l'abonnement
                    // Ici on laisse le bouton copier libre
                }
            }
            type();
        }

        function copyCode() {
            const text = terminal.innerText;
            navigator.clipboard.writeText(text).then(() => {
                showToast("Code copié !");
            });
        }

        function closeModal() {
            document.getElementById('subscription-modal').classList.remove('active');
        }

        function showToast(msg) {
            const t = document.getElementById('toast');
            t.innerText = msg;
            t.style.display = 'block';
            setTimeout(() => t.style.display = 'none', 3000);
        }
    </script>
